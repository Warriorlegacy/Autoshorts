import { GoogleGenerativeAI } from '@google/generative-ai';
import fs from 'fs';
import path from 'path';
import https from 'https';
import { promisify } from 'util';
 
export interface ImageGenerationOptions {
  prompt: string;
  style?: 'cinematic' | 'animated' | 'minimalist' | 'documentary' | 'stock';
  aspectRatio?: '9:16' | '16:9' | '1:1';
  quality?: 'low' | 'medium' | 'high';
}
 
export interface ImageGenerationResponse {
  imageUrl: string;
  localPath: string;
  prompt: string;
  style: string;
  generatedAt: Date;
}
 
/**
 * Image Generation Service
 * Uses Gemini API to generate background images for video scenes
 */
export class ImageGenerationService {
  private genAI: GoogleGenerativeAI | null;
  private imagesDir: string;
  private mockMode: boolean;
 
  constructor() {
    const apiKey = process.env.GEMINI_API_KEY;
 
    if (!apiKey) {
      console.warn('‚ö†Ô∏è Gemini API key not found. Image generation will use mock mode.');
      this.mockMode = true;
      this.genAI = null;
    } else {
      console.log('üé® Gemini Image Generation: Initialized with real API key');
      this.mockMode = false;
      this.genAI = new GoogleGenerativeAI(apiKey);
    }
 
    // Ensure images directory exists
    this.imagesDir = path.join(process.cwd(), 'public', 'images');
    if (!fs.existsSync(this.imagesDir)) {
      fs.mkdirSync(this.imagesDir, { recursive: true });
    }
    
    // Schedule periodic cleanup of old images (every 24 hours)
    this.scheduleCleanup();
  }
 
  /**
   * Schedule periodic cleanup of old images
   */
  private scheduleCleanup(): void {
    // Run cleanup every 24 hours
    setInterval(() => {
      this.cleanupOldImages(7).catch((error) => {
        console.error('Scheduled cleanup failed:', error);
      });
    }, 24 * 60 * 60 * 1000);
  }
 
  /**
   * Generate an image based on a prompt and style
   */
  async generateImage(options: ImageGenerationOptions): Promise<ImageGenerationResponse> {
    if (this.mockMode || !this.genAI) {
      return this.generateMockImage(options);
    }
 
    try {
      // Enhanced prompt with style instructions
      const enhancedPrompt = this.buildPrompt(options);
 
      console.log(`üé® Generating image: "${options.prompt}" (${options.style})`);
 
      // Use the vision model for image generation
      const model = this.genAI.getGenerativeModel({ model: 'gemini-pro-vision' });
 
      // Note: Gemini doesn't have direct image generation, so we'll use gemini-pro
      // to generate a detailed image prompt, then use it as a reference
      const textModel = this.genAI.getGenerativeModel({ model: 'gemini-pro' });
      const result = await textModel.generateContent(enhancedPrompt);
      const response = await result.response;
      const text = response.text();
 
      console.log(`‚úÖ Generated image prompt: ${text.substring(0, 100)}...`);
 
      // For now, we'll return a mock image as Gemini doesn't directly generate images
      // In production, you would use an image generation API or stock image service
      return this.generateMockImage(options);
    } catch (error) {
      console.error('Error generating image:', error);
      return this.generateMockImage(options);
    }
  }
 
  /**
   * Generate multiple images in batch
   */
  async generateBatch(
    prompts: string[],
    style: 'cinematic' | 'animated' | 'minimalist' | 'documentary' | 'stock' = 'cinematic'
  ): Promise<ImageGenerationResponse[]> {
    try {
      const results = await Promise.all(
        prompts.map((prompt) =>
          this.generateImage({
            prompt,
            style,
            aspectRatio: '9:16',
            quality: 'high',
          })
        )
      );
 
      console.log(`‚úÖ Generated ${results.length} images in batch`);
      return results;
    } catch (error) {
      console.error('Error in batch image generation:', error);
      return [];
    }
  }
 
  /**
   * Get stock image suggestions using Gemini
   */
  async getStockImageSuggestions(niche: string, sceneDescription: string): Promise<string[]> {
    if (!this.genAI) {
      return this.getMockStockSuggestions(niche);
    }
 
    try {
      const prompt = `Suggest 3 stock image search keywords for a ${niche} video scene: "${sceneDescription}". 
      Format as a JSON array of strings.
      Example: ["technology innovation", "digital transformation", "futuristic business"]`;
 
      const model = this.genAI.getGenerativeModel({ model: 'gemini-pro' });
      const result = await model.generateContent(prompt);
      const response = await result.response;
      const text = response.text();
 
      try {
        // Extract JSON from response
        const jsonMatch = text.match(/\[[\s\S]*\]/);
        if (jsonMatch) {
          const suggestions = JSON.parse(jsonMatch[0]);
          return suggestions;
        }
      } catch (e) {
        console.warn('Failed to parse stock image suggestions:', e);
      }
 
      return this.getMockStockSuggestions(niche);
    } catch (error) {
      console.error('Error getting stock suggestions:', error);
      return this.getMockStockSuggestions(niche);
    }
  }
 
  /**
   * Build enhanced prompt for image generation
   */
  private buildPrompt(options: ImageGenerationOptions): string {
    const styleGuides: Record<string, string> = {
      cinematic: 'Cinematic, professional film quality, high contrast, dramatic lighting, 4K resolution',
      animated: 'Vibrant animated style, motion graphics, dynamic composition, colorful, modern design',
      minimalist: 'Minimalist, clean lines, simple color palette, modern aesthetic, uncluttered',
      documentary: 'Documentary style, educational, professional photography, realistic, informative',
      stock: 'Stock photo quality, professional, suitable for commercial use, high resolution',
    };
 
    const styleGuide = styleGuides[options.style || 'cinematic'];
    const aspectRatio = options.aspectRatio || '9:16';
 
    return `Generate a detailed image prompt for: "${options.prompt}"
    
Style: ${styleGuide}
Aspect Ratio: ${aspectRatio}
Quality: ${options.quality || 'high'}
 
The prompt should be suitable for image generation APIs. Include specific visual elements, lighting, mood, and technical details.`;
  }
 
  /**
   * Generate mock image (for development)
   */
  private async generateMockImage(options: ImageGenerationOptions): Promise<ImageGenerationResponse> {
    // Create a gradient SVG as a placeholder image
    const colors = this.getGradientForStyle(options.style || 'cinematic');
    const svgContent = `
    <svg width="1080" height="1920" xmlns="http://www.w3.org/2000/svg">
      <defs>
        <linearGradient id="grad" x1="0%" y1="0%" x2="100%" y2="100%">
          <stop offset="0%" style="stop-color:${colors.start};stop-opacity:1" />
          <stop offset="100%" style="stop-color:${colors.end};stop-opacity:1" />
        </linearGradient>
      </defs>
      <rect width="1080" height="1920" fill="url(#grad)"/>
      <text x="540" y="960" font-size="48" fill="white" text-anchor="middle" font-family="Arial">
        ${options.prompt.substring(0, 30)}
      </text>
      <text x="540" y="1020" font-size="24" fill="white" opacity="0.7" text-anchor="middle" font-family="Arial">
        ${options.style} ‚Ä¢ Mock Generated
      </text>
    </svg>
    `;
 
    const filename = `image_mock_${Date.now()}_${Math.random().toString(36).substring(7)}.svg`;
    const filepath = path.join(this.imagesDir, filename);
 
    await promisify(fs.writeFile)(filepath, svgContent);
 
    console.log(`üé® Mock image created: ${filename}`);
 
    return {
      imageUrl: `/images/${filename}`,
      localPath: filepath,
      prompt: options.prompt,
      style: options.style || 'cinematic',
      generatedAt: new Date(),
    };
  }
 
  /**
   * Get gradient colors based on style
   */
  private getGradientForStyle(
    style: 'cinematic' | 'animated' | 'minimalist' | 'documentary' | 'stock'
  ): { start: string; end: string } {
    const gradients: Record<string, { start: string; end: string }> = {
      cinematic: { start: '#0f0c29', end: '#302b63' },
      animated: { start: '#FF6B6B', end: '#4ECDC4' },
      minimalist: { start: '#F5F5F5', end: '#CCCCCC' },
      documentary: { start: '#2C3E50', end: '#34495E' },
      stock: { start: '#667EEA', end: '#764BA2' },
    };
 
    return gradients[style] || gradients.cinematic;
  }
 
  /**
   * Get mock stock suggestions
   */
  private getMockStockSuggestions(niche: string): string[] {
    const suggestions: Record<string, string[]> = {
      Technology: ['technology innovation', 'digital transformation', 'futuristic', 'coding', 'software'],
      Gaming: ['gaming competition', 'esports', 'console gaming', 'game streaming', 'player profile'],
      Education: ['classroom learning', 'online education', 'studying', 'knowledge', 'academic'],
      Comedy: ['funny moment', 'comedy show', 'laugh', 'humor', 'entertainment'],
      Fitness: ['workout training', 'gym exercise', 'healthy lifestyle', 'athletic', 'sports'],
      Business: ['corporate meeting', 'business growth', 'entrepreneurship', 'success', 'teamwork'],
      Travel: ['travel destination', 'adventure', 'exploration', 'tourism', 'landscape'],
      Food: ['food preparation', 'cooking', 'recipe', 'culinary', 'ingredients'],
      Health: ['wellness', 'meditation', 'healthy living', 'nutrition', 'healthcare'],
      Finance: ['financial planning', 'investment', 'wealth', 'money', 'trading'],
      Lifestyle: ['daily life', 'living well', 'style', 'fashion', 'personal growth'],
      Entertainment: ['movie scene', 'entertainment', 'drama', 'performance', 'show'],
    };
 
    return suggestions[niche] || suggestions.Technology;
  }
 
  /**
   * Clean up old images (older than 7 days)
   */
  async cleanupOldImages(maxAgeDays = 7): Promise<number> {
    try {
      // Use async file operations instead of sync
      const files = await promisify(fs.readdir)(this.imagesDir);
      const now = Date.now();
      let deletedCount = 0;
 
      // Process files concurrently with Promise.all for better performance
      await Promise.all(
        files.map(async (file) => {
          if (file.startsWith('image_') && (file.endsWith('.svg') || file.endsWith('.jpg') || file.endsWith('.png'))) {
            const filepath = path.join(this.imagesDir, file);
            try {
              const stats = await promisify(fs.stat)(filepath);
              const ageDays = (now - stats.mtimeMs) / (1000 * 60 * 60 * 24);
 
              if (ageDays > maxAgeDays) {
                await promisify(fs.unlink)(filepath);
                deletedCount++;
              }
            } catch (error) {
              console.warn(`Failed to cleanup file ${file}:`, error);
              // Continue with next file
            }
          }
        })
      );
 
      if (deletedCount > 0) {
        console.log(`üóëÔ∏è Cleaned up ${deletedCount} old images`);
      }
 
      return deletedCount;
    } catch (error) {
      console.error('Error cleaning up images:', error);
      return 0;
    }
  }
}
 
// Singleton instance
let imageService: ImageGenerationService | null = null;
 
export function getImageGenerationService(): ImageGenerationService {
  if (!imageService) {
    imageService = new ImageGenerationService();
  }
  return imageService;
}
